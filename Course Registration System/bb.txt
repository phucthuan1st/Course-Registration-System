bool writeToFile_info(string file  student  l) {
	fstream f;
	f.open(file  ios::out);

			f << l.data.No << "\n";
			f << l.data.studentID << "\n";
			f << l.data.firstName << "\n";
			f << l.data.lastName << "\n";
			f << l.data.gender << "\n";
			f << l.data.date.day << "\n";
			f << l.data.date.month << "\n";
			f << l.data.date.year << "\n";
			f << l.data.socialID << "\n";
		
	f.close();
	return true;
}
